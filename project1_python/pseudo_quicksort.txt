Quicksort(A, p, r)
1 if p < r
2 // Partition the subarray around the pivot, which ends up in A[q].
3 q = Partition(A, p, r)
4 Quicksort(A, p, q − 1) // recursively sort the low side
5 Quicksort(A, q + 1, r) // recursively sort the low side
Partition(A, p, r)
1 x = A[r]
2 i = p − 1
3 for j = p to r − 1
4 if A[j] ≤ x
5 i = i + 1
6 exchange A[i] with A[j]
7 exchange A[i = 1] with A[j]
8 return i + 1
